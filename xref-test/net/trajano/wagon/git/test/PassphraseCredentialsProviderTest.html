11  public class PassphraseCredentialsProviderTest {
12      @Test
13      public void testCredentials() throws Exception {
14          final CredentialItem.StringType i = new CredentialItem.StringType(
15                  "enter passphrase", true);
16          final CredentialItem i2 = new CredentialItem.InformationalMessage(
17                  "info");
18          final PassphraseCredentialsProvider provider = new PassphraseCredentialsProvider(
19                  "passphrase");
20          assertTrue(provider.get(null, i, i2));
21          assertEquals("passphrase", i.getValue());
22          assertFalse(provider.isInteractive());
23          assertTrue(provider.supports(i, i2));
24          assertTrue(provider.supports(i));
25          assertFalse(provider.supports(i2));
26          assertFalse(provider.supports());
27      }
28  
29      @Test
30      public void testEmptyCredentials() throws Exception {
31          final PassphraseCredentialsProvider provider = new PassphraseCredentialsProvider(
32                  "passphrase");
33          assertTrue(provider.get(null));
34          assertFalse(provider.supports());
35      }
